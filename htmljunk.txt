   function setCookie(name, value, expires, path, domain, secure) {
var today = new Date();
today.setTime( today.getTime() );
if ( expires )
{
expires = expires * 1000 * 60 * 60 * 24;
}
var expires_date = new Date( today.getTime() + (expires) );
document.cookie = name + "=" +escape( value ) +
( ( expires ) ? ";expires=" + expires_date.toGMTString() : "" ) + //expires.toGMTString()
( ( path ) ? ";path=" + path : "" ) +
( ( domain ) ? ";domain=" + domain : "" ) +
( ( secure ) ? ";secure" : "" );
}
function getCookie(name) {
var dc = document.cookie;
var prefix = name + "=";
var begin = dc.indexOf("; " + prefix);
if (begin == -1) {
begin = dc.indexOf(prefix);
if (begin != 0) return null;
} else {
begin += 2;
}
var end = document.cookie.indexOf(";", begin);
if (end == -1) {
end = dc.length;
}
return unescape(dc.substring(begin + prefix.length, end));
}

function toCurrency(num) {
var sign;
var i;
var plus;
num = num.toString().replace(/\$|\,/g, '');
if (isNaN(num)) {
num = "0";
}
sign = (num == (num = Math.abs(num)));
num = Math.floor(num * 100 + 0.50000000001);
num = Math.floor(num / 100).toString();
for (i = 0; i < Math.floor((num.length - (1 + i)) / 3); i++) {
if(num == 500000){
plus = "+";
}else{
plus = "";
}
num = num.substring(0, num.length - (4 * i + 3)) + 'K' + plus;
}
return (((sign) ? '' : '') + '$' + num);
}

function drawMarkersVE() {
var marker;
mainLayer = new Microsoft.Maps.EntityCollection({visible:true});
infoboxCollection = new Microsoft.Maps.EntityCollection({visible:true});
var prevMarkerIndex = null;
var firstMarkerIndex = null;
for ( var c = 0; c < markersData.length; c++ ) {
var md = markersData[c];
if (md.lat != "") {
if (firstMarkerIndex == null) {
firstMarkerIndex = c;
}
md.prevMarkerIndex = prevMarkerIndex;
if (prevMarkerIndex != null) {
markersData[prevMarkerIndex].nextMarkerIndex = c;
}
var opt = {icon: getcolorIcons(md.listingTypeName), width:10, height:15, typeName: 'my_pushpin'};
marker = new Microsoft.Maps.Pushpin(new Microsoft.Maps.Location(parseFloat(md.lat), parseFloat(md.lng)), opt);
marker.mdindex = c;
md.markerVE = marker;
mainLayer.push(marker);
} else {
mainLayer.push(new Microsoft.Maps.Pushpin(new Microsoft.Maps.Location(0,0)));
}
}
map.entities.push(mainLayer);
for ( var j = 0; j < markersData.length; j++ ) {
var md = markersData[j];
if ( md.lat != "" ) {
marker = md.markerVE;
var info = new Microsoft.Maps.Infobox(marker.getLocation(), { visible: false, offset: infoOffset, showCloseButton:false, htmlContent: createListingInfo(j, false), showPointer: false});
infoboxCollection.push(info);
} else {
infoboxCollection.push(new Microsoft.Maps.Infobox(new Microsoft.Maps.Location(0,0)));
}
}
map.entities.push(infoboxCollection);
if(markersData.length == 0){
map.setView({zoom:3});
}
}

function updatingMarkersVE(view, updateMarkersData, mainTotal){
map.entities.clear();
if(mainTotal == 0){
return false;
}
var marker;
mainLayer = new Microsoft.Maps.EntityCollection({visible:true});
infoboxCollection = new Microsoft.Maps.EntityCollection({visible:true});
var prevMarkerIndex = null;
var firstMarkerIndex = null;
var maxLat = null;
var maxLot = null;
var latitude = null;
var longitude = null;
var geocodesFound = false;
for ( var c = 0; c < updateMarkersData.length; c++ ) {
var md = updateMarkersData[c];
if(md.lat == "") {
$j.extend(updateMarkersData[c],{"skipNext":"1"});
}
if (md.lat != "") {
if (firstMarkerIndex == null) {
firstMarkerIndex = c;
}
latitude = md.lat;
longitude = md.lng;
if(geocodesFound == false){
maxLat = latitude;
maxLot = longitude;
minLat = latitude;
minLot = longitude;
geocodesFound = true
}
if(latitude >= maxLat){
maxLat = md.lat;
}
if(longitude >= maxLot){
maxLot = md.lng;
}
if(latitude <= minLat){
minLat = md.lat;
}
if(longitude <= minLot){
minLot = md.lng;
}
var opt = {icon: getcolorIcons(md.listingTypeName), width:10, height:15, typeName: 'my_pushpin'};
marker = new Microsoft.Maps.Pushpin(new Microsoft.Maps.Location(parseFloat(md.lat), parseFloat(md.lng)), opt);
marker.mdindex = c;
md.markerVE = marker;
mainLayer.push(marker);
} else {
mainLayer.push(new Microsoft.Maps.Pushpin(new Microsoft.Maps.Location(0,0)));
}
}
map.entities.push(mainLayer);
for ( var j = 0; j < updateMarkersData.length; j++ ) {
var md = updateMarkersData[j];
if ( md.lat != "" ) {
marker = md.markerVE;
var info = new Microsoft.Maps.Infobox(marker.getLocation(), { visible: false, offset: infoOffset, showCloseButton:false, htmlContent: createListingInfo(j, true), showPointer: false});
infoboxCollection.push(info);
} else {
infoboxCollection.push(new Microsoft.Maps.Infobox(new Microsoft.Maps.Location(0,0)));
}
}
map.entities.push(infoboxCollection);
var TopLeftLatLong = new Microsoft.Maps.Location(maxLat, minLot);
var BottomRightLatLong = new Microsoft.Maps.Location(minLat, maxLot);
var BoundingRectangle = new Microsoft.Maps.LocationRect.fromCorners(TopLeftLatLong, BottomRightLatLong);
map.setView({bounds:BoundingRectangle});
//map.setView({center:new Microsoft.Maps.Location(maxLat, maxLot)});
}
